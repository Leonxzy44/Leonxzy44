<script>
// Flappy Grok - Fully playable in GitHub README! ğŸ¦œ
const canvas = document.getElementById('canvas');
const ctx = canvas.getContext('2d');
let birdY = 250, velocity = 0, gravity = 0.5, lift = -10, score = 0, gameRunning = false;
let pipes = [], pipeGap = 150, pipeSpeed = 3, pipeFreq = 90, frame = 0;
const highScore = localStorage.getItem('flappyGrokHS') || 0;
document.getElementById('high-score').textContent = `Best: ${highScore}`;

function drawBird() {
  ctx.fillStyle = '#ffeb3b';
  ctx.beginPath();
  ctx.arc(80, birdY, 20, 0, Math.PI*2);
  ctx.fill();
  // eye
  ctx.fillStyle = 'white';
  ctx.beginPath();
  ctx.arc(90, birdY-8, 8, 0, Math.PI*2);
  ctx.fill();
  ctx.fillStyle = 'black';
  ctx.beginPath();
  ctx.arc(92, birdY-8, 4, 0, Math.PI*2);
  ctx.fill();
  // beak
  ctx.fillStyle = '#ff9800';
  ctx.beginPath();
  ctx.moveTo(100, birdY);
  ctx.lineTo(120, birdY);
  ctx.lineTo(100, birdY+10);
  ctx.fill();
}

function drawPipes() {
  pipes.forEach(p => {
    // top pipe
    ctx.fillStyle = '#4caf50';
    ctx.fillRect(p.x, 0, 60, p.top);
    ctx.fillStyle = '#388e3c';
    ctx.fillRect(p.x-5, p.top-40, 70, 40);
    // bottom pipe
    ctx.fillStyle = '#4caf50';
    ctx.fillRect(p.x, p.top + pipeGap, 60, canvas.height);
    ctx.fillStyle = '#388e3c';
    ctx.fillRect(p.x-5, p.top + pipeGap, 70, 40);
  });
}

function update() {
  if (!gameRunning) return;
  velocity += gravity;
  birdY += velocity;
  
  if (birdY > canvas.height - 30 || birdY < 20) endGame();
  
  if (frame % pipeFreq === 0) {
    const top = Math.random() * (canvas.height - pipeGap - 200) + 50;
    pipes.push({x: canvas.width, top});
  }
  
  pipes.forEach((p,i) => {
    p.x -= pipeSpeed;
    if (p.x + 60 < 80 && !p.scored) {
      score++;
      p.scored = true;
      new Audio('data:audio/wav;base64,UklGRiQAAABXQVZFZm10IBAAAAABAAEARKwAAIhYAQACABAAZGF0YQAAAAA=')?.play();
    }
    if (p.x < 60 && p.x + 60 > 60) {
      if (birdY - 20 < p.top || birdY + 20 > p.top + pipeGap) endGame();
    }
    if (p.x < -100) pipes.splice(i,1);
  });
  
  frame++;
  draw();
  requestAnimationFrame(update);
}

function draw() {
  ctx.clearRect(0,0,canvas.width,canvas.height);
  // sky gradient
  const grad = ctx.createLinearGradient(0,0,0,canvas.height);
  grad.addColorStop(0, '#87CEEB');
  grad.addColorStop(1, '#E0F7FA');
  ctx.fillStyle = grad;
  ctx.fillRect(0,0,canvas.width,canvas.height);
  
  drawPipes();
  drawBird();
  
  ctx.fillStyle = 'white';
  ctx.font = 'bold 50px Arial';
  ctx.strokeStyle = 'black';
  ctx.lineWidth = 4;
  ctx.strokeText(score, canvas.width/2-25, 80);
  ctx.fillText(score, canvas.width/2-25, 80);
}

function endGame() {
  gameRunning = false;
  document.getElementById('game-over').style.display = 'flex';
  document.getElementById('final-score').textContent = `Score: ${score}`;
  if (score > highScore) {
    localStorage.setItem('flappyGrokHS', score);
    document.getElementById('high-score').textContent = `Best: ${score} ğŸ† NEW RECORD!`;
  }
}

function startGame() {
  birdY = 250; velocity = 0; score = 0; pipes = []; frame = 0; gameRunning = true;
  update();
}

canvas.addEventListener('click', () => { if(gameRunning) velocity = lift; });
canvas.addEventListener('touchstart', e => { e.preventDefault(); if(gameRunning) velocity = lift; });
document.addEventListener('keydown', e => { if(e.code === 'Space' && gameRunning) velocity = lift; });
</script>
